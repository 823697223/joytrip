# confirm

const Confirm: (message: string, title?: string) => Promise<boolean>

参数

  1. message 提示信息
  2. title 提示标题

example

<template>
	<div>
		<el-button type="primary" @click="open">打开Confirm</el-button>
	</div>
</template>

<script setup lang="ts">
import { Confirm, Message } from '@joytrip/apsat'

const open = async () => {
	await Confirm('是否确认?')
	// 如果点击取消，不会执行下面逻辑
	Message.success('确认')
}
</script>

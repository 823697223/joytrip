<template>
	<Card style="height: 600px; overflow: auto">
		<ListContainer title="标题" shadow="Title">
			<template #title-btns>
				<el-button type="primary">标题右侧按钮</el-button>
			</template>
			<template #condition>
				<ConditionRender :condition="condition" :items="items">
					<template #btns>
						<ResetButton></ResetButton>
						<SearchButton></SearchButton>
					</template>
				</ConditionRender>
			</template>
			<template #middle>搜索条件与列表之间插槽</template>
			<TableRender :options="options"></TableRender>
			<template #page>
				<Pagination :total="0" :currentPage="1"></Pagination>
			</template>
		</ListContainer>
	</Card>
</template>

<script setup lang="ts">
import { ConditionItems, TableOptions, createInput, createSelect } from '@joytrip/vue3'
import { computed, ref } from 'vue'

const condition = ref({
	input: '',
	select: '',
})
const items = computed<ConditionItems>(() => ({
	ticketNo: {
		label: 'input',
		components: createInput({}),
	},
	status: {
		label: 'select',
		components: createSelect({
			options: [],
		}),
	},
}))

const options = computed<TableOptions<any>>(() => {
	return {
		data: new Array(20).fill({ name: 'name', decs: 'desc' }),
		columns: [
			{
				label: '属性名',
				value: 'name',
			},
			{
				label: '说明',
				value: 'desc',
			},
		],
	}
})
</script>

<style scoped lang="scss">
::v-deep(.page) {
	padding: 0;
}
</style>

<template>
	<Card>
		<Step style="padding: 20px" :steps="steps" ref="stepRef"> </Step>
		<div>
			<el-button type="primary" @click="prev">上一步</el-button>
			<el-button type="primary" @click="next">下一步</el-button>
		</div>
	</Card>
</template>

<script setup lang="ts">
import { StepItem, useStep, Step } from '@joytrip/apsat'
import { ref } from 'vue'

const steps = ref<StepItem[]>([{ name: '第一步' }, { name: '第二步' }, { name: '第三步' }])
const { next, prev } = useStep()
</script>

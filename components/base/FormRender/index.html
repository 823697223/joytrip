Input 输入框
Select 选择器
DatePicker 日期选择器
Switch 开关
Radio 单选框
Checkbox 多选框
InputNumber 计数器
Button 按钮
Button 按钮组
render 自定义内容
我是自定义内容
设置样式
不用el-form-item包裹的内容，一般用于分割线
<template>
	<Card>
		<FormRender :options="options"></FormRender>
	</Card>
</template>

<script setup lang="ts">
import {
	FormOptions,
	createButton,
	createCheckbox,
	createDatePicker,
	createInput,
	createInputNumber,
	createRadio,
	createSelect,
	createSwitch,
} from '@joytrip/vue3'
import { h } from 'vue'
import { computed, ref } from 'vue'

const form = ref({
	input: 'input',
	select: '',
	datePicker: '',
	switch: true,
	radio: '1',
	checkbox: true,
	inputNumber: 0,
})
const buttons = ref<('default' | 'primary' | 'success' | 'warning' | 'info' | 'danger')[]>([
	'default',
	'primary',
	'success',
	'warning',
	'danger',
	'info',
])
const options = computed<FormOptions>(() => {
	return {
		// el-form的model属性
		model: form,
		// 表单项，items对象的key默认等同于form的key，也可以使用无关的items的key，但是需要通过设置prop来与form进行绑定
		items: {
			input: {
				label: 'Input 输入框',
				components: createInput({}),
			},
			select: {
				label: 'Select 选择器',
				components: createSelect({
					options: [
						{
							label: 'Restaurant',
							value: '1',
						},
						{
							label: 'Order No.',
							value: '2',
							disabled: true,
						},
						{
							label: 'Group',
							options: [
								{
									label: 'Group1',
									value: '3',
								},
								{
									label: 'Group2',
									value: '4',
								},
							],
						},
					],
				}),
			},
			datePicker: {
				label: 'DatePicker 日期选择器',
				components: createDatePicker({}),
			},
			switch: {
				label: 'Switch 开关',
				components: createSwitch({}),
			},
			radio: {
				label: 'Radio 单选框',
				components: createRadio({
					options: [
						{
							label: 'Restaurant',
							value: '1',
						},
						{
							label: 'Order No.',
							value: '2',
							disabled: true,
						},
					],
				}),
			},
			checkbox: {
				label: 'Checkbox 多选框',
				components: createCheckbox({}),
			},
			inputNumber: {
				label: 'InputNumber 计数器',
				components: createInputNumber({}),
			},
			button: {
				label: 'Button 按钮',
				components: createButton({
					content: 'button',
				}),
			},
			buttons: {
				label: 'Button 按钮组',
				components: buttons.value.map((type, index) =>
					createButton({
						content: 'button' + (index + 1),
						type,
					})
				),
			},
			render: {
				label: 'render 自定义内容',
				render: () => h('div', '我是自定义内容'),
			},
			renderColor: {
				label: '设置样式',
				components: createInput({
					placeholder: '300px的input',
					// 组件样式
					style: {
						width: '300px',
					},
				}),
				// el-form-item样式
				style: {
					background: 'red',
				},
			},
			hideFormItem: {
				render: () => h('div', '不用el-form-item包裹的内容，一般用于分割线'),
				hideFormItem: true,
			},
		},
		labelWidth: 160,
	}
})
</script>

# 表单校验

Input 输入框（使用required属性）
Select 选择器（使用rules属性）
DatePicker 日期选择器（自定义validator校验器）
<template>
	<Card>
		<div class="row">
			<FormRender :options="options" ref="formRef"></FormRender>
		</div>
		<div class="row">
			<el-button type="primary" @click="validateForm">校验表单</el-button>
			<el-button type="primary" @click="resetForm">重置表单</el-button>
		</div>
	</Card>
</template>

<script setup lang="ts">
import { FormOptions, createDatePicker, createInput, createSelect, useForm } from '@joytrip/vue3'
import { ElMessage } from 'element-plus'
import { computed, ref } from 'vue'

const { validate, resetFields } = useForm()
const formRef = ref()
const form = ref({
	datePicker: '',
})
const validateForm = async () => {
	if (!(await validate())) {
		ElMessage.error('validate error')
		return
	}
	ElMessage.success('validate success')
}
const resetForm = () => resetFields()
const options = computed<FormOptions>(() => {
	return {
		model: form,
		items: {
			// 表单必填项，createInput默认的trigger是blur，其他的是change
			input: {
				label: 'Input 输入框（使用required属性）',
				required: true,
				components: createInput({}),
			},
			// 使用原始rules属性
			select: {
				label: 'Select 选择器（使用rules属性）',
				rules: [
					{
						required: true,
						message: '请选择',
						blur: 'change',
					},
				],
				components: createSelect({
					options: [
						{
							label: 'Restaurant',
							value: '1',
						},
						{
							label: 'Order No.',
							value: '2',
							disabled: true,
						},
						{
							label: 'Group',
							options: [
								{
									label: 'Group1',
									value: '3',
								},
								{
									label: 'Group2',
									value: '4',
								},
							],
						},
					],
				}),
			},
			// 自定义validator校验器
			datePicker: {
				label: 'DatePicker 日期选择器（自定义validator校验器）',
				required: true,
				validator(rule, value, callback) {
					if (!value) callback('请选择日期')
					callback()
				},
				components: createDatePicker({}),
			},
		},
		labelPosition: 'top',
	}
})
</script>

# 表单组件使用其他数据

默认使用model
ref数据
对象的属性
对象的嵌套属性
数组元素
<template>
	<Card>
		<FormRender :options="options"></FormRender>
	</Card>
</template>

<script setup lang="ts">
import { FormOptions, createInput } from '@joytrip/vue3'
import { computed, ref } from 'vue'

const form = ref({
	input: 'input',
})
const input2 = ref('input2')
const input3 = ref({
	input: 'input3',
	input4: {
		input: 'input4',
	},
})
const array = ref([1, 2, 3, 4, 5])
const options = computed<FormOptions>(() => {
	return {
		model: form,
		items: {
			input: {
				label: '默认使用model',
				components: createInput({
					// data默认使用model对应的数据
					// data: form.value,
					// dataKey默认使用当前对象的key即 input
					// dataKey: 'input',
				}),
			},
			input2: {
				label: 'ref数据',
				components: createInput({
					// input2组件绑定input2变量
					data: input2,
					// 如果data是ref，dataKey默认为value
					// dataKey: 'value',
				}),
			},
			input3: {
				label: '对象的属性',
				components: createInput({
					// input3组件绑定input3的input
					data: input3.value,
					dataKey: 'input',
				}),
			},
			input4: {
				label: '对象的嵌套属性',
				components: createInput({
					// input4组件绑定input3.input4.input
					data: input3.value.input4,
					dataKey: 'input',
				}),
			},
			input5: {
				label: '数组元素',
				components: createInput({
					// input5组件绑定array数组的某个元素
					data: array.value,
					dataKey: '2',
				}),
			},
		},
		labelWidth: 160,
	}
})
</script>

# 使用修饰符

Input 输入框
<template>
	<Card>
		<FormRender :options="options"></FormRender>
	</Card>
</template>

<script setup lang="ts">
import { FormOptions, createButton, createInput } from '@joytrip/vue3'
import { h } from 'vue'
import { computed, ref } from 'vue'

const form = ref({
	input: 'input',
})
const show = ref(true)
const options = computed<FormOptions>(() => {
	return {
		model: form,
		items: {
			input: {
				label: 'Input 输入框',
				components: createInput({
					type: 'number',
					if: show.value,
					modifier: {
						// 等同于 v-model.number
						number: true,
					},
				}),
			},
			button: {
				label: '',
				components: createButton({
					type: 'primary',
					content: show.value ? '隐藏Input' : '显示Input',
					onClick() {
						show.value = !show.value
					},
				}),
			},
		},
		labelWidth: 160,
	}
})
</script>

<template>
	<Card>
		<FormRender :options="options"></FormRender>
	</Card>
</template>

<script setup lang="ts">
import { FormOptions, createText } from '@joytrip/vue3'
import { computed, ref } from 'vue'

const form = ref({})
const options = computed<FormOptions>(() => {
	return {
		model: form,
		items: {
			text: {
				label: 'text',
				components: createText({
					text: 'text内容',
				}),
			},
		},
	}
})
</script>

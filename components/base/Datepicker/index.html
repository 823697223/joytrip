# DatePicker 日期选择器

至
<template>
	<Card>
		<div class="row">
			<DatePicker v-model="date" ref="datepickerRef"></DatePicker>
		</div>
		<div class="row">
			<el-button type="primary" @click="focus">点击获得焦点</el-button>
			<el-button type="primary" @click="handle('focus')">使用通用方法-点击获得焦点</el-button>
		</div>
		<div class="row">
			<DatePicker
				v-model="date1"
				type="daterange"
				ref="datepickerRef"
				:disabledDate="disabledDate"
			></DatePicker>
		</div>
	</Card>
</template>

<script setup lang="ts">
import { ref } from 'vue'

const date = ref('')
const date1 = ref('')
const datepickerRef = ref()
const focus = () => {
	datepickerRef.value.focus()
}
// 调用组件内部的组件的通用方法
const handle = (methodName: string) => {
	datepickerRef.value.handle(methodName)
}
const disabledDate = (time: Date) => {
	return time.getTime() > Date.now()
}
</script>

import { renderDatePicker } from '@joytrip/vue3'
import { ref } from 'vue'

const date = ref('')
const render = () => {
	return renderDatePicker({
		data: date,
		dataKey: 'value',
	})
}
</script>

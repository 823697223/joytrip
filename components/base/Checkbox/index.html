# Checkbox 多选框

Checkbox 多选框
Checkbox 多选框组
<template>
	<Card>
		<FormRender :options="options"></FormRender>
	</Card>
</template>

<script setup lang="ts">
import { FormOptions, createCheckbox } from '@joytrip/vue3'
import { computed, ref } from 'vue'

const form = ref({
	checkbox: true,
	checkboxs: [],
})
const options = computed<FormOptions>(() => {
	return {
		model: form,
		items: {
			checkbox: {
				label: 'Checkbox 多选框',
				components: createCheckbox({}),
			},
			checkboxs: {
				label: 'Checkbox 多选框组',
				components: createCheckbox({
					options: [
						{
							label: '选项1',
							value: 1,
						},
						{
							label: '选项2',
							value: 2,
						},
						{
							label: '选项3',
							value: 3,
						},
					],
				}),
			},
		},
		labelWidth: 160,
	}
})
</script>

# render

<template>
	<Card>
		<render :render="render"></render>
	</Card>
</template>

<script setup lang="ts">
import { renderCheckbox } from '@joytrip/vue3'
import { ref } from 'vue'

const form = ref({
	checkbox: true,
	checkboxs: [],
})
const render = () => {
	return renderCheckbox({
		data: form.value,
		dataKey: 'checkbox',
		label: '选项',
	})
}
</script>

# Divider 分割线

# 基础用法

<template>
	<Card>
		<FormRender :options="options"></FormRender>
	</Card>
</template>

<script setup lang="ts">
import { FormOptions, createDivider } from '@joytrip/vue3'
import { computed, ref } from 'vue'

const form = ref({})
const options = computed<FormOptions>(() => {
	return {
		model: form,
		items: {
			line: {
				hideFormItem: true,
				components: createDivider({}),
			},
		},
	}
})
</script>

# render

<template>
	<Card>
		<render :render="render"></render>
	</Card>
</template>

<script setup lang="ts">
import { renderDivider } from '@joytrip/vue3'

const render = () => {
	return renderDivider({})
}
</script>

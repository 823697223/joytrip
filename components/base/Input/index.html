# Input 输入框

<template>
	<Card>
		<div class="row">
			<Input v-model="input" ref="inputRef"></Input>
		</div>
		<div class="row">
			<el-button type="primary" @click="focus">点击获得焦点</el-button>
			<el-button type="primary" @click="handle('focus')">使用通用方法-点击获得焦点</el-button>
		</div>
		<div class="row">
			<Input v-model="input1" disabled></Input>
		</div>
		<div class="row">
			<Input v-model="input2" type="password"></Input>
		</div>
		<div class="row">
			<Input v-model="input3" type="textarea"></Input>
		</div>
		<div class="row">
			<Input
				v-model="input4"
				:formatter="(value) => `$ ${value}`.replace(/\B(?=(\d{3})+(?!\d))/g, ',')"
				:parser="(value) => value.replace(/\$\s?|(,*)/g, '')"
			></Input>
		</div>
		<div class="row">
			<Input v-model="input5" :prefix-icon="Search" :suffix-icon="Search"></Input>
		</div>
		<div class="row">
			<Input v-model="input6">
				<template #prepend>
					<Select
						v-model="select"
						placeholder="Select"
						:options="options"
						style="width: 115px"
					>
					</Select>
				</template>
				<template #append>
					<el-button :icon="Search" />
				</template>
			</Input>
		</div>
	</Card>
</template>

<script setup lang="ts">
import { ref } from 'vue'
import { Search } from '@element-plus/icons-vue'
import { reactive } from 'vue'

const input = ref('')
const input1 = ref('')
const input2 = ref('')
const input3 = ref('')
const input4 = ref('')
const input5 = ref('')
const input6 = ref('')
const select = ref('')
const options = reactive([
	{
		label: 'Restaurant',
		value: '1',
	},
	{
		label: 'Order No.',
		value: '2',
	},
	{
		label: 'Tel',
		value: '3',
	},
])
const inputRef = ref()
const focus = () => {
	inputRef.value.focus()
}
// 调用组件内部的组件的通用方法
const handle = (methodName: string) => {
	inputRef.value.handle(methodName)
}
</script>

<template>
	<Card>
		<render :render="render"></render>
	</Card>
</template>

<script setup lang="ts">
import { renderInput } from '@joytrip/vue3'
import { ref } from 'vue'

const input = ref('')
const render = () => {
	return renderInput({
		data: input,
		dataKey: 'value',
	})
}
</script>

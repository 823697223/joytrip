# TableRender 表格渲染器 ⭐

# 基础用法

<template>
	<Card>
		<div class="row">
			<TableRender :options="options" ref="tableRef"></TableRender>
		</div>
		<div class="row">
			<el-button type="primary" @click="select">选中第一行</el-button>
		</div>
	</Card>
</template>

<script setup lang="ts">
import { TableOptions, useTableSelection } from '@joytrip/vue3'
import { computed, ref } from 'vue'

interface Data {
	name: string
	description: string
	managementGroupName: string
}

const { onSelectionChange } = useTableSelection()
const data = ref<Data[]>([
	{
		name: 'name',
		description: 'description',
		managementGroupName: 'managementGroupName',
	},
])
const tableRef = ref()
const select = () => {
	tableRef.value.handle('toggleRowSelection', data.value[0])
}
const options = computed<TableOptions<Data>>(() => ({
	data: data.value,
	columns: [
		{
			type: 'selection',
		},
		{
			label: '模板名称',
			value: 'name',
			fixed: true,
		},
		{
			label: '模板描述',
			value: 'description',
			showTooltipWhenOverflow: true,
		},
		{
			label: '所属管理组',
			value: 'managementGroupName',
		},
	],
	onSelectionChange,
}))
</script>

# 编辑 row

<template>
	<Card>
		<div class="row">
			<TableRender :options="options" ref="tableRef"></TableRender>
		</div>
		<div class="row">
			<el-button type="primary" @click="editRow(data[0], true)">编辑第一行</el-button>
			<el-button type="primary" @click="editRow(data[0], false)">保存第一行</el-button>
			<el-button type="primary" @click="validateTable">校验表格</el-button>
		</div>
	</Card>
</template>

<script setup lang="ts">
import { Message } from '@joytrip/apsat'
import { TableOptions, renderInput, useTable } from '@joytrip/vue3'
import { computed, h, ref, resolveComponent } from 'vue'

interface Data {
	name: string
	description: string
	managementGroupName: string
}

const { editRow, validate } = useTable()
const validateTable = async () => {
	if (await validate()) {
		Message.success('校验成功')
	} else {
		Message.error('校验失败')
	}
}

const data = ref<Data[]>([
	{
		name: 'name',
		description: 'description',
		managementGroupName: 'managementGroup',
	},
	{
		name: 'name',
		description: 'description',
		managementGroupName: 'managementGroup',
	},
	{
		name: 'name',
		description: 'description',
		managementGroupName: 'managementGroup',
	},
])
const tableRef = ref()
const options = computed<TableOptions<Data>>(() => ({
	data: data.value,
	columns: [
		{
			type: 'selection',
		},
		{
			label: '模板名称',
			value: 'name',
			// 简单用法
			edit: {
				// type 表示组件类型
				type: 'input',
				// 必填
				required: true,
			},
		},
		{
			label: '模板描述',
			value: 'description',
			// 简单用法
			edit: {
				// type 表示组件类型
				type: 'select',
				options: [{ label: '1', value: '1' }],
				// 自定义校验
				validator: (r, v, c) => {
					c()
				},
			},
		},
		{
			label: '所属管理组',
			value: 'managementGroupName',
			// 高级用法
			edit: {
				render: (scope) =>
					renderInput({
						// 必须设置data和dataKey 否则没有响应式
						data: scope.row,
						dataKey: 'managementGroupName',
					}),
			},
		},
		{
			label: '操作',
			render: ({ row }) => {
				return [
					h(
						resolveComponent('el-button'),
						{
							type: 'primary',
							onClick: () => {
								tableRef.value.editRow(row, true)
							},
						},
						() => '编辑'
					),
					h(
						resolveComponent('el-button'),
						{
							type: 'primary',
							onClick: () => {
								tableRef.value.editRow(row, false)
							},
						},
						() => '保存'
					),
				]
			},
		},
	],
}))
</script>

<script setup lang="ts">
import { TableOptions, renderTable, useTableSelection } from '@joytrip/vue3'
import { computed, ref } from 'vue'

interface Data {
	name: string
	description: string
	managementGroupName: string
}

const { onSelectionChange } = useTableSelection()
const data = ref<Data[]>([
	{
		name: 'name',
		description: 'description',
		managementGroupName: 'managementGroupName',
	},
])

const options = computed<TableOptions<Data>>(() => ({
	data: data.value,
	columns: [
		{
			type: 'selection',
		},
		{
			label: '模板名称',
			value: 'name',
			fixed: true,
		},
		{
			label: '模板描述',
			value: 'description',
			showTooltipWhenOverflow: true,
		},
		{
			label: '所属管理组',
			value: 'managementGroupName',
		},
	],
	onSelectionChange,
}))

const render = () => {
	return renderTable(options.value)
}
</script>

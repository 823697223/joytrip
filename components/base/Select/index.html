# Select 选择器

# 基础用法

<template>
	<Card>
		<div class="row">
			<Select
				v-model="select"
				:options="options"
				placeholder="请选择"
				ref="selectRef"
			></Select>
		</div>
		<div class="row">
			<el-button type="primary" @click="focus">点击获得焦点</el-button>
			<el-button type="primary" @click="handle('focus')">使用通用方法-点击获得焦点</el-button>
		</div>
		<div class="row">
			<Select v-model="select2" :options="options" disabled></Select>
		</div>
		<div class="row">
			<Select v-model="select3" :options="options" placeholder="请选择" filterable></Select>
		</div>
		<div class="row">
			<Select
				v-model="select4"
				:options="options"
				placeholder="请选择"
				:optionsRender="optionsRender"
			></Select>
		</div>
	</Card>
</template>

<script setup lang="ts">
import { h, ref } from 'vue'
import { reactive } from 'vue'
import { SelectOptions } from '@joytrip/vue3'

const select = ref('')
const select2 = ref('select2')
const select3 = ref('')
const select4 = ref('')
const options = reactive<SelectOptions[]>([
	{
		label: 'Restaurant',
		value: '1',
	},
	{
		label: 'Order No.',
		value: '2',
		disabled: true,
	},
	{
		label: 'Group',
		options: [
			{
				label: 'Group1',
				value: '3',
			},
			{
				label: 'Group2',
				value: '4',
			},
		],
	},
])
const optionsRender = (option: SelectOptions) => {
	return h('div', [h('span', option.label), h('span', '+' + option.value)])
}
const selectRef = ref()
const focus = () => {
	selectRef.value.focus()
}
// 调用组件内部的组件的通用方法
const handle = (methodName: string) => {
	selectRef.value.handle(methodName)
}
</script>

# render

<template>
	<Card>
		<render :render="render"></render>
	</Card>
</template>

<script setup lang="ts">
import { SelectOptions, renderSelect } from '@joytrip/vue3'
import { reactive, ref } from 'vue'

const select = ref('')
const options = reactive<SelectOptions[]>([
	{
		label: 'Restaurant',
		value: '1',
	},
	{
		label: 'Order No.',
		value: '2',
		disabled: true,
	},
	{
		label: 'Group',
		options: [
			{
				label: 'Group1',
				value: '3',
			},
			{
				label: 'Group2',
				value: '4',
			},
		],
	},
])
const render = () => {
	return renderSelect({
		data: select,
		dataKey: 'value',
		options,
	})
}
</script>

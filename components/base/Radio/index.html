# Radio 单选框

Radio 多选框组
<template>
	<Card>
		<FormRender :options="options"></FormRender>
	</Card>
</template>

<script setup lang="ts">
import { FormOptions, createRadio } from '@joytrip/vue3'
import { computed, ref } from 'vue'

const form = ref({
	radio: null,
})
const options = computed<FormOptions>(() => {
	return {
		model: form,
		items: {
			radio: {
				label: 'Radio 多选框组',
				components: createRadio({
					options: [
						{
							label: '选项1',
							value: 1,
						},
						{
							label: '选项2',
							value: 2,
						},
						{
							label: '选项3',
							value: 3,
						},
					],
				}),
			},
		},
		labelWidth: 160,
	}
})
</script>

# render

<template>
	<Card>
		<render :render="render"></render>
	</Card>
</template>

<script setup lang="ts">
import { renderRadio } from '@joytrip/vue3'
import { ref } from 'vue'

const form = ref({
	radios: null,
})
const render = () => {
	return renderRadio({
		data: form.value,
		dataKey: 'radios',
		options: [
			{
				label: '选项1',
				value: 1,
			},
			{
				label: '选项2',
				value: 2,
			},
			{
				label: '选项3',
				value: 3,
			},
		],
	})
}
</script>
